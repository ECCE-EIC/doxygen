42 // Class description :
43 //
44 // G4UIXm : class to handle a Motif interactive session.
45 // G4UIXm is the Motif version of G4UIterminal.
46 //
47 // A command box is at disposal for entering/recalling Geant4 commands.
48 // A menubar could be customized through the AddMenu, AddButton methods.
49 // Note that there are corresponding Geant4 commands to add a
50 // menus in the menubar and add buttons in a menu.
51 // Ex :
52 // /gui/addMenu test Test
53 // /gui/addButton test Init /run/initialize
54 // /gui/addButton test "Set gun" "/control/execute gun.g4m"
55 // /gui/addButton test "Run one event" "/run/beamOn 1"
56 //
57 // Command completion, by typing "tab" key, is available on the
58 // command line.
59 //
60 // Class description - end :
61 
62 class G4UIXm : public G4VBasicShell, public G4VInteractiveSession {
63 public: // With description
64  G4UIXm(int,char**);
65  // (argv, argc) or (0, NULL) had to be given.
66  G4UIsession* SessionStart();
67  // To enter interactive X loop ; waiting/executing command,...
68  void AddMenu(const char*,const char*);
69  // To add a pulldown menu in the menu bar.
70  // First argument is the name of the menu.
71  // Second argument is the label of the cascade button.
72  // Ex : AddMenu("my_menu","My menu")
73  void AddButton(const char*,const char*,const char*);
74  // To add a push button in a pulldown menu.
75  // First argument is the name of the menu.
76  // Second argument is the label of the button.
77  // Third argument is the Geant4 command executed when the button is fired.
78  // Ex : AddButton("my_menu","Run","/run/beamOn 1");
