59 G4double CexmcStudiedProcess::PostStepGetPhysicalInteractionLength(
60  const G4Track & track, G4double , G4ForceCondition * condition )
61 {
62  *condition = NotForced;
63 
64  if ( ! physicsManager->IsStudiedProcessAllowed() )
65  return CexmcDblMax;
66 
67  CexmcTrackInfo * trackInfo( static_cast< CexmcTrackInfo * >(
68  track.GetUserInformation() ) );
69 
70  if ( ! trackInfo ||
71  trackInfo->GetTypeInfo() != CexmcIncidentParticleTrackType )
72  return CexmcDblMax;
73 
74  CexmcIncidentParticleTrackInfo * theTrackInfo(
75  static_cast< CexmcIncidentParticleTrackInfo * >( trackInfo ) );
76 
77  if ( ! theTrackInfo->IsStudiedProcessActivated() )
78  return CexmcDblMax;
79 
80  return theTrackInfo->GetFinalTrackLengthInTarget() -
81  theTrackInfo->GetCurrentTrackLengthInTarget();
82 }
