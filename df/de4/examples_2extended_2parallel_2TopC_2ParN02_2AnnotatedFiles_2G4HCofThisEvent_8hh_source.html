40 // class description:
41 //
42 // This is a class which stores hits collections generated at one event.
43 // This class is exclusively constructed by G4SDManager when the first
44 // hits collection of an event is passed to the manager, and this class
45 // object is deleted by G4RunManager when a G4Event class object is deleted.
46 // Almost all public methods must be used by Geant4 kernel classes and
47 // the user should not invoke them. The user can use two const methods,
48 // GetHC() and GetNumberOfCollections() for accessing to the stored hits
49 // collection(s).

75  std::vector<G4VHitsCollection*> * HC; /*MSH: ptr_as_array
76  [elementType: G4VHitsCollection*]
77  [elementCount: { $ELE_COUNT = $THIS->GetNumberOfCollections(); }]
78  [elementGet: { $ELEMENT = $THIS->GetHC($ELE_INDEX); }]

80 
81  public: // with description
82  inline G4VHitsCollection* GetHC(G4int i)
83  { return (*HC)[i]; }
84  // Returns a pointer to a hits collection. Null will be returned
85  // if the particular collection is not stored at the current event.
86  // The integer argument is ID number which is assigned by G4SDManager
87  // and the number can be obtained by G4SDManager::GetHitsCollectionID()
88  // method.
89  inline G4int GetNumberOfCollections()
90  {
91  G4int n = 0;
92  for(size_t i=0;i<HC->size();i++)
93  {
94  if((*HC)[i]) n++;
95  }
96  return n;
97  }
98  // Returns the number of hits collections which are stored in this class
99  // object.
100  public:
101  inline G4int GetCapacity()
102  {
103  return HC->size();
104  }
105 };
