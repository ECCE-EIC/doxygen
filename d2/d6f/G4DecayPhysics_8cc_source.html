99 void G4DecayPhysics::ConstructProcess()
100 {
101  if(wasActivated) { return; }
102  wasActivated = true;
103 
104  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();
105 
106  // Add Decay Process
107  fDecayProcess = new G4Decay();
108  auto myParticleIterator=GetParticleIterator();
109  myParticleIterator->reset();
110  G4ParticleDefinition* particle=0;
111 
112  while( (*myParticleIterator)() )
113  {
114  particle = myParticleIterator->value();
115  if( fDecayProcess->IsApplicable(*particle) )
116  {
117  if(verbose > 1) {
118  G4cout << "### Decays for " << particle->GetParticleName() << G4endl;
119  }
120  ph->RegisterProcess(fDecayProcess, particle);
121  }
122  }
123 }
