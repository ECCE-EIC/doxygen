133 inline
134  G4int G4ExcitedDeltaConstructor::GetQuarkContents(G4int iQ, G4int iIso3)
135 {
136  // Quark contents
137  // iIso3 = +3 : uuu
138  // iIso3 = +1 : uud
139  // iIso3 = -1 : udd
140  // iIso3 = -3 : ddd
141  G4int quark=0;
142  if ( iQ == 0 ){
143  if ( iIso3 == -3 ){
144  // d-quark
145  quark = 1;
146  } else {
147  // u-quark
148  quark = 2;
149  }
150  } else if ( iQ == 2 ){
151  if ( iIso3 == +3 ){
152  // u-quark
153  quark = 2;
154  } else {
155  // d-quark
156  quark = 1;
157  }
158  } else {
159  if (( iIso3 == -1 )||( iIso3 == -3 )) {
160  // d-quark
161  quark = 1;
162  } else {
163  // u-quark
164  quark = 2;
165  }
166  }
167  return quark;
168 }
