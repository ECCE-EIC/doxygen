141 inline void DetectorConstruction::ComputeCalorParameters()
142 {
143  // Compute derived parameters of the calorimeter
144  fLayerThickness = fAbsorberThickness + fGapThickness;
145  fCalorThickness = fNbOfLayers*fLayerThickness;
146 
147  fWorldSizeX = 1.2*fCalorThickness; fWorldSizeYZ = 1.2*fCalorSizeYZ;
148 }
