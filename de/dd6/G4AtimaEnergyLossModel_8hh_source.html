189 inline void G4AtimaEnergyLossModel::SetParticle(const G4ParticleDefinition* p)
190 {
191  if(particle != p) {
192  particle = p;
193  if(p->GetBaryonNumber() > 3 || p->GetPDGCharge() > CLHEP::eplus)
194  { isIon = true; }
195  SetupParameters();
196  }
197 }
