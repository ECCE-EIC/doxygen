60 PhysicsList::PhysicsList()
61  : G4VModularPhysicsList(),
62  fEmDNAPhysicsList(nullptr),fEmDNAChemistryList(nullptr),
63  fEmDNAChemistryList1(nullptr),fPhysDNAName("")
64 {
65  G4double currentDefaultCut = 1.*nanometer;
66  // fixe lower limit for cut
67  G4ProductionCutsTable::GetProductionCutsTable()->
68  SetEnergyRange(100*eV, 1*GeV);
69  SetDefaultCutValue(currentDefaultCut);
70  SetVerboseLevel(1);
71 
72  RegisterConstructor("G4EmDNAPhysics");
73  if(CommandLineParser::GetParser()->GetCommandIfActive("-chemOFF")==0)
74  {
75  RegisterConstructor("G4EmDNAChemistry");
76  }
77 
78 }

110 void PhysicsList::RegisterConstructor(const G4String& name)
111 {
112  if(name == fPhysDNAName) { return; }
113  if(verboseLevel > 0) {
114  G4cout << "===== Register constructor ==== " << name << G4endl;
115  }
116  if(name == "G4EmDNAPhysics") {
117  delete fEmDNAPhysicsList;
118  fEmDNAPhysicsList = new G4EmDNAPhysics(verboseLevel);
119  fPhysDNAName = name;
120  } else if(name == "G4EmDNAPhysics_option1") {
121  delete fEmDNAPhysicsList;
122  fEmDNAPhysicsList = new G4EmDNAPhysics_option1(verboseLevel);
123  fPhysDNAName = name;
124  } else if(name == "G4EmDNAPhysics_option2") {
125  delete fEmDNAPhysicsList;
126  fEmDNAPhysicsList = new G4EmDNAPhysics_option2(verboseLevel);
127  fPhysDNAName = name;
128  } else if(name == "G4EmDNAPhysics_option3") {
129  delete fEmDNAPhysicsList;
130  fEmDNAPhysicsList = new G4EmDNAPhysics_option3(verboseLevel);
131  fPhysDNAName = name;
132  } else if(name == "G4EmDNAPhysics_option4") {
133  delete fEmDNAPhysicsList;
134  fEmDNAPhysicsList = new G4EmDNAPhysics_option4(verboseLevel);
135  fPhysDNAName = name;
136  } else if(name == "G4EmDNAPhysics_option5") {
137  delete fEmDNAPhysicsList;
138  fEmDNAPhysicsList = new G4EmDNAPhysics_option5(verboseLevel);
139  fPhysDNAName = name;
140  } else if(name == "G4EmDNAPhysics_option6") {
141  delete fEmDNAPhysicsList;
142  fEmDNAPhysicsList = new G4EmDNAPhysics_option6(verboseLevel);
143  fPhysDNAName = name;
144  } else if(name == "G4EmDNAPhysics_option7") {
145  delete fEmDNAPhysicsList;
146  fEmDNAPhysicsList = new G4EmDNAPhysics_option7(verboseLevel);
147  fPhysDNAName = name;
148  } else if(name == "G4EmDNAPhysics_option8") {
149  delete fEmDNAPhysicsList;
150  fEmDNAPhysicsList = new G4EmDNAPhysics_option8(verboseLevel);
151  fPhysDNAName = name;
152  } else if(name == "G4EmDNAChemistry") {
153  if(fEmDNAChemistryList || fEmDNAChemistryList1) { return; }
154  fEmDNAChemistryList = new G4EmDNAChemistry();
155  fEmDNAChemistryList->SetVerboseLevel(verboseLevel);
156  } else if(name == "G4EmDNAChemistry_option1") {
157  if(fEmDNAChemistryList || fEmDNAChemistryList1) { return; }
158  fEmDNAChemistryList1 = new G4EmDNAChemistry_option1();
159  fEmDNAChemistryList1->SetVerboseLevel(verboseLevel);
160  } else {
161  G4cout << "PhysicsList::RegisterConstructor: <" << name << ">"
162  << " fails - name is not defined"
163  << G4endl;
164  }
165 }
