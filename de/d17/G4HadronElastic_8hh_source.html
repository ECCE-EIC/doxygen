100 inline G4double
101 G4HadronElastic::ComputeMomentumCMS(const G4ParticleDefinition* p,
102  G4double plab, G4int Z, G4int A)
103 {
104  G4double m1 = p->GetPDGMass();
105  G4double m12= m1*m1;
106  G4double mass2 = G4NucleiProperties::GetNuclearMass(A, Z);
107  return plab*mass2/std::sqrt(m12 + mass2*mass2 + 2.*mass2*std::sqrt(m12 + plab*plab));
108 }
