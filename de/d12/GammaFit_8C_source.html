7 Double_t GammaFunction(Double_t* t, Double_t* params ) {
8  if (ROOT::TMath::Gamma(params[0]) != 0 ) {
9  return (100 * pow(params[1],params[0]) / ROOT::TMath::Gamma(params[0]))
10  * pow(t[0],(params[0]-1)) * exp (-params[1]*t[0]);
11  } else {
12  return 0;
