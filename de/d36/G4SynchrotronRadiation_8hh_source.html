113 inline G4double
114 G4SynchrotronRadiation::Chebyshev(G4double a, G4double b, const G4double c[],
115  G4int n, G4double x)
116 {
117  G4double y;
118  G4double y2=2.0*(y=(2.0*x-a-b)/(b-a)); // Change of variable.
119  G4double d=0.,dd=0.;
120  for (G4int j=n-1;j>=1;--j) // Clenshaw's recurrence.
121  { G4double sv=d;
122  d=y2*d-dd+c[j];
123  dd=sv;
124  }
125  return y*d-dd+0.5*c[0];
126 }
