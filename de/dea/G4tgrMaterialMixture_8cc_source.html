54 G4tgrMaterialMixture::G4tgrMaterialMixture(const G4String& matType,
55  const std::vector<G4String>& wl)
56 {
57  //---------- Check for miminum number of words read
58  G4tgrUtils::CheckWLsize( wl, 6, WLSIZE_GE,
59  "G4tgrMaterialMixture::G4tgrMaterialMixture" );
60 
61  theMateType = matType;
62 
63  //---------- Fill private data
64  theName = G4tgrUtils::GetString( wl[1] );
65  theDensity = std::fabs(G4tgrUtils::GetDouble( wl[2], g/cm3 ) );
66  theNoComponents = G4tgrUtils::GetInt( wl[3] );
67 
68  G4tgrUtils::CheckWLsize( wl, 4+theNoComponents*2, WLSIZE_GE,
69  "G4tgrMaterialMixture::G4tgrMaterialMixture" );
70  for(G4int ii=0; ii<theNoComponents; ii++)
71  {
72 #ifdef G4VERBOSE
73  if( G4tgrMessenger::GetVerboseLevel() >= 3 )
74  {
75  G4cout << " G4tgrMaterialMixture::G4tgrMaterialMixture() -"
76  << " adding component: " << wl[ii*2+4] << " Fraction= "
77  << G4tgrUtils::GetDouble(wl[ii*2+1+4]) << G4endl;
78  }
79 #endif
80  theComponents.push_back( G4tgrUtils::GetString( wl[ii*2+4] ) );
81  theFractions.push_back( G4tgrUtils::GetDouble(wl[ii*2+1+4]) );
82  }
83 
84 #ifdef G4VERBOSE
85  if( G4tgrMessenger::GetVerboseLevel() >= 1 )
86  {
87  G4cout << " Created " << *this << G4endl;
88  }
89 #endif
90 }

103  }
