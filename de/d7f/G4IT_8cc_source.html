139 void G4IT::TakeOutBox()
140 {
141  if(fpITBox)
142  {
143  fpITBox->Extract(this);
144  fpITBox = nullptr;
145  }
146 
147  if(fpTrackNode)
148  {
149  delete fpTrackNode;
150  fpTrackNode = nullptr;
151  }
152 
153  if(fpKDNode)
154  {
155  InactiveNode(fpKDNode);
156  fpKDNode = nullptr;
157  }
158 }

180 G4bool G4IT::operator<(const G4IT& right) const
181 {
182  if (GetITType() == right.GetITType())
183  {
184  return (this->diff(right));
185  }
186  else
187  {
188  return (GetITType() < right.GetITType());
189  }
190  return false;
191 }
