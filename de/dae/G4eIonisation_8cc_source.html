92 void G4eIonisation::InitialiseEnergyLossProcess(
93  const G4ParticleDefinition* part,
94  const G4ParticleDefinition*)
95 {
96  if(!isInitialised) {
97  if(part != theElectron) { isElectron = false; }
98  if (!EmModel(0)) { SetEmModel(new G4MollerBhabhaModel()); }
99  G4EmParameters* param = G4EmParameters::Instance();
100  EmModel(0)->SetLowEnergyLimit(param->MinKinEnergy());
101  EmModel(0)->SetHighEnergyLimit(param->MaxKinEnergy());
102  if (!FluctModel()) { SetFluctModel(new G4UniversalFluctuation()); }
103 
104  AddEmModel(1, EmModel(), FluctModel());
105  isInitialised = true;
106  }
107 }
