208  // Visualization attributes with opacity!

302  fsomaS [i] = new G4Orb("Soma", fNeuronLoadParamz->GetRadSomacomp(i)* um);

304  // you can change parameters of Soma with a single compartment
305  fsomaLV[i] = new G4LogicalVolume(fsomaS[i], fpWaterMaterial, "Soma");
306  fsomaLV[i] ->SetVisAttributes(fSomaColour);
307  fsomaPV[i] = new G4PVPlacement(
308  0, // no rotation
309  G4ThreeVector(
310  (fNeuronLoadParamz->GetPosSomacomp(i).x()-fNeuronLoadParamz->GetshiftX())*um,
311  (fNeuronLoadParamz->GetPosSomacomp(i).y()-fNeuronLoadParamz->GetshiftY())*um,
312  (fNeuronLoadParamz->GetPosSomacomp(i).z()-fNeuronLoadParamz->GetshiftZ())*um),
313  fsomaLV[i],
314  "Soma",
315  boundingLV,
316  false,
317  i);
318  fMassSomacomp[i] = fNeuronLoadParamz->GetMassSomacomp(i) ;
319  fPosSomacomp[i] = fNeuronLoadParamz->GetPosSomacomp(i) ;
320  fpRegion->SetProductionCuts(cuts);
