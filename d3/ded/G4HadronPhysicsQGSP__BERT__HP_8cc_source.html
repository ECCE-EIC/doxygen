88 void G4HadronPhysicsQGSP_BERT_HP::Neutron()
89 {
90  auto neu = new G4NeutronBuilder( true ); // Fission on
91  AddBuilder(neu);
92  auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
93  AddBuilder(qgs);
94  qgs->SetMinEnergy(minQGSP_neutron);
95  neu->RegisterMe(qgs);
96  auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
97  AddBuilder(ftf);
98  ftf->SetMinEnergy(minFTFP_neutron);
99  ftf->SetMaxEnergy(maxFTFP_neutron);
100  neu->RegisterMe(ftf);
101  auto bert = new G4BertiniNeutronBuilder;
102  AddBuilder(bert);
103  bert->SetMinEnergy(minBERT_neutron);
104  bert->SetMaxEnergy(maxBERT_neutron);
105  neu->RegisterMe(bert);
106  auto hp = new G4NeutronPHPBuilder;
107  AddBuilder(hp);
108  neu->RegisterMe(hp);
109  neu->Build();
110 }

112 void G4HadronPhysicsQGSP_BERT_HP::ExtraConfiguration()
113 {
114  // --- Neutrons ---
115  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
116  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
117  if (capture) {
118  G4NeutronRadCapture* theNeutronRadCapture = new G4NeutronRadCapture();
119  theNeutronRadCapture->SetMinEnergy( minBERT_neutron );
120  capture->RegisterMe( theNeutronRadCapture );
121  }
122  G4HadronicProcess* fission = G4PhysListUtil::FindFissionProcess(neutron);
123  if (fission) {
124  G4LFission* theNeutronLEPFission = new G4LFission();
125  theNeutronLEPFission->SetMinEnergy( minBERT_neutron );
126  theNeutronLEPFission->SetMaxEnergy( G4HadronicParameters::Instance()->GetMaxEnergy() );
127  fission->RegisterMe( theNeutronLEPFission );
128  }
129 }
