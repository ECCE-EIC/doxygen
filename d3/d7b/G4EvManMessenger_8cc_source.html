36 G4EvManMessenger::G4EvManMessenger(G4EventManager * fEvMan)
37 :fEvManager(fEvMan)
38 {
39  eventDirectory = new G4UIdirectory("/event/");
40  eventDirectory->SetGuidance("EventManager control commands.");
41 
42  abortCmd = new G4UIcmdWithoutParameter("/event/abort",this);
43  abortCmd->SetGuidance("Abort current event.");
44  abortCmd->AvailableForStates(G4State_EventProc);
45 
46  verboseCmd = new G4UIcmdWithAnInteger("/event/verbose",this);
47  verboseCmd->SetGuidance("Set Verbose level of event management category.");
48  verboseCmd->SetGuidance(" 0 : Silent");
49  verboseCmd->SetGuidance(" 1 : Stacking information");
50  verboseCmd->SetGuidance(" 2 : More...");
51  verboseCmd->SetParameterName("level",false);
52  verboseCmd->SetRange("level>=0");
53 
54  storeEvtCmd = new G4UIcmdWithoutParameter("/event/keepCurrentEvent",this);
55  storeEvtCmd->SetGuidance("Store the current event to G4Run object instead of deleting it at the end of event.");
56  storeEvtCmd->SetGuidance("Stored event is available through G4Run until the beginning of next run.");
57  storeEvtCmd->SetGuidance("Given the potential large memory size of G4Event and its data-member objects stored in G4Event,");
58  storeEvtCmd->SetGuidance("the user must be careful and responsible for not to store too many G4Event objects.");
59  storeEvtCmd->AvailableForStates(G4State_EventProc);
60 }

70 void G4EvManMessenger::SetNewValue(G4UIcommand * command,G4String newValues)
71 {
72  if( command == verboseCmd )
73  { fEvManager->SetVerboseLevel(verboseCmd->GetNewIntValue(newValues)); }
74  if( command == abortCmd )
75  { fEvManager->AbortCurrentEvent(); }
76  if( command == storeEvtCmd )
77  { fEvManager->KeepTheCurrentEvent(); }
78 }

80 G4String G4EvManMessenger::GetCurrentValue(G4UIcommand * command)
81 {
82  G4String cv;
83  if( command == verboseCmd )
84  { cv = verboseCmd->ConvertToString(fEvManager->GetVerboseLevel()); }
85  return cv;
86 }
