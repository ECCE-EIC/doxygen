46 G4BcMesonPlus* G4BcMesonPlus::Definition()
47 {
48  if (theInstance !=0) return theInstance;
49  const G4String name = "Bc+";
50  // search in particle table]
51  G4ParticleTable* pTable = G4ParticleTable::GetParticleTable();
52  G4ParticleDefinition* anInstance = pTable->FindParticle(name);
53  if (anInstance ==0)
54  {
55  // create particle
56  //
57  // Arguments for constructor are as follows
58  // name mass width charge
59  // 2*spin parity C-conjugation
60  // 2*Isospin 2*Isospin3 G-parity
61  // type lepton number baryon number PDG encoding
62  // stable lifetime decay table
63  // shortlived subType anti_encoding
64 
65  anInstance = new G4ParticleDefinition(
66  name, 6.2749*GeV, 1.298e-9*MeV, +1.*eplus,
67  0, -1, 0,
68  0, 0, 0,
69  "meson", 0, 0, 541,
70  false, 0.507e-3*ns, NULL,
71  false, "Bc");
72  }
73  theInstance = reinterpret_cast<G4BcMesonPlus*>(anInstance);
74  return theInstance;
75 }
