162 inline void G4eBremParametrizedModel::SetCurrentElement(const G4double Z)
163 {
164  //std::cout<<"SetCurrentElement Z="<<Z<<std::endl;
165  if(Z != currentZ) {
166  currentZ = Z;
167 
168  G4int iz = G4int(Z);
169  z13 = nist->GetZ13(iz);
170  z23 = z13*z13;
171  lnZ = nist->GetLOGZ(iz);
172 
173  Fel = facFel - lnZ/3. ;
174  Finel = facFinel - 2.*lnZ/3. ;
175 
176  fCoulomb = GetCurrentElement()->GetfCoulomb();
177  fMax = Fel-fCoulomb + Finel/currentZ + (1.+1./currentZ)/12.;
178 
179  }
180 }
