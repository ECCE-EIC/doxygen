58 void SteppingAction::UserSteppingAction(const G4Step* step)
59 {
60 
61  G4double edep = step->GetTotalEnergyDeposit();
62 
63  //total energy deposit
64  fEventAction->AddEdep(edep);
65 
66  if (step->GetTrack()->GetDynamicParticle()->GetDefinition()
67  == G4Electron::ElectronDefinition())
68  {
69  const G4String& processName = step->GetPostStepPoint()->
70  GetProcessDefinedStep()->GetProcessName();
71 
72  if (processName!="Transportation")
73  {
74  // get analysis manager
75  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
76 
77  // fill histo
78 
79  // 1) ALL e-
80  // analysisManager
81  // ->FillH1(1,
82  // step->GetTrack()->GetKineticEnergy()/eV,step->GetStepLength()/nm);
83  analysisManager->FillH1(1,
84  step->GetPreStepPoint()->GetKineticEnergy()/eV,
85  step->GetStepLength()/nm);
86 
87  // 2) PARENT e-
88  if (step->GetTrack()->GetParentID()==0
89  && step->GetTrack()->GetTrackID()==1)
90  // analysisManager->FillH1(2, step->GetTrack()->GetKineticEnergy()/eV,
91  // step->GetStepLength()/nm);
92  analysisManager->FillH1(2,
93  step->GetPreStepPoint()->GetKineticEnergy()/eV,
94  step->GetStepLength()/nm);
95 
96  // 3) SECONDARY e-
97  if ( !(step->GetTrack()->GetParentID()==0)
98  || !(step->GetTrack()->GetTrackID()==1) )
99  // analysisManager->FillH1(3, step->GetTrack()->GetKineticEnergy()/eV,
100  // step->GetStepLength()/nm);
101  analysisManager->FillH1(3,
102  step->GetPreStepPoint()->GetKineticEnergy()/eV,
103  step->GetStepLength()/nm);
104 
105  }
106 
107  }
108 }
