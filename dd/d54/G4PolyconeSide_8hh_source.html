30 // Class implmenting a face that represents one conical side
31 // of a polycone:
32 //
33 // G4PolyconeSide( const G4PolyconeSideRZ *prevRZ,
34 // const G4PolyconeSideRZ *tail,

38 // G4bool phiIsOpen, G4bool isAllBehind=false )
