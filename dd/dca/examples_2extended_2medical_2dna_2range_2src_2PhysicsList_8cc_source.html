192 {
193  G4PhysicsListHelper* ph = G4PhysicsListHelper::GetPhysicsListHelper();

195  auto particleIterator=GetParticleIterator();
196  particleIterator->reset();
197  while ((*particleIterator)())
198  {
199  G4ParticleDefinition* particle = particleIterator->value();
