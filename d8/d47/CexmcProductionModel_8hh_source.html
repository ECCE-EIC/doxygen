244 inline G4bool CexmcProductionModel::IsValidCandidateForAngularRange(
245  G4double top, G4double bottom, G4int nmbOfDivs ) const
246 {
247  return top > bottom && top <= 1.0 && top > -1.0 && bottom < 1.0 &&
248  bottom >= -1.0 && nmbOfDivs >= 1;
