378 inline void CexmcReconstructor::TransformToAdjacentInnerCrystal(
379  G4int & column, G4int & row ) const
380 {
381  if ( column == 0 )
382  ++column;
383  if ( column == calorimeterGeometry.nCrystalsInRow - 1 )
384  --column;
385  if ( row == 0 )
386  ++row;
387  if ( row == calorimeterGeometry.nCrystalsInColumn - 1 )
388  --row;
389 }
