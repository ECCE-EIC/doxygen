56 G4tgrPlaceDivRep::G4tgrPlaceDivRep( const std::vector<G4String>& wl )
57 {
58  theDivType = DivByNdivAndWidth;
59 
60  // Name parent axis nrep width offset
61  G4tgrUtils::CheckWLsize( wl, 6, WLSIZE_GE,
62  "G4tgrPlaceDivRep::G4tgrPlaceDivRep" );
63  G4tgrUtils::CheckWLsize( wl, 7, WLSIZE_LE,
64  "G4tgrPlaceDivRep::G4tgrPlaceDivRep" );
65 
66  theParentName = G4tgrUtils::GetString(wl[2]);
67  theAxis = BuildAxis( G4tgrUtils::GetString(wl[3]) );
68  theNDiv = G4tgrUtils::GetInt( wl[4] );
69  theWidth = G4tgrUtils::GetDouble(wl[5])*mm; // check if it is deg
70  if( wl.size() == 7 )
71  {
72  theOffset = G4tgrUtils::GetDouble(wl[6])*mm; // check if it is deg
73  }
74  else
75  {
76  theOffset = 0.;
77  }
78 
79 #ifdef G4VERBOSE
80  if( G4tgrMessenger::GetVerboseLevel() >= 1 )
81  {
82  G4cout << " Created " << *this << G4endl;
83  }
84 #endif
85 
86 }

90 EAxis G4tgrPlaceDivRep::BuildAxis( const G4String& axisName )
91 {
92  if( axisName == "X" ) {
93  return kXAxis;
94  } else if( axisName == "Y" ) {
95  return kYAxis;
96  } else if( axisName == "Z" ) {
97  return kZAxis;
98  } else if( axisName == "R" ) {
99  return kRho;
100  } else if( axisName == "PHI" ) {
101  return kPhi;
102  }
103  else
104  {
105  G4String ErrMessage = "Axis type not found: " + axisName
106  + ". Only valid axis are: X, Y, Z, R, PHI !";
107  G4Exception("G4tgrVolumeDivision::GetReplicaAxis()",
108  "InvalidAxis", FatalException, ErrMessage);
109  }
110  return kXAxis; // to avoid warning errors
111 }
