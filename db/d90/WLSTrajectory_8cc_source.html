84 WLSTrajectory::WLSTrajectory(WLSTrajectory & right) : G4VTrajectory()

139 void WLSTrajectory::MergeTrajectory(G4VTrajectory* secondTrajectory)
140 {
141  if(!secondTrajectory) return;
142 
143  WLSTrajectory* second = (WLSTrajectory*)secondTrajectory;
144  G4int ent = second->GetPointEntries();
145  // initial point of the second trajectory should not be merged
146  for(G4int i=1; i<ent; ++i) {
147  fpPointsContainer->push_back((*(second->fpPointsContainer))[i]);
148  }
149  delete (*second->fpPointsContainer)[0];
150  second->fpPointsContainer->clear();
151 }
