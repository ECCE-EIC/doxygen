128 G4Polyhedron* G4CSGSolid::GetPolyhedron () const
129 {
130  if (fpPolyhedron == nullptr ||
131  fRebuildPolyhedron ||
132  fpPolyhedron->GetNumberOfRotationStepsAtTimeOfCreation() !=
133  fpPolyhedron->GetNumberOfRotationSteps())
134  {
135  G4AutoLock l(&polyhedronMutex);
136  delete fpPolyhedron;
137  fpPolyhedron = CreatePolyhedron();
138  fRebuildPolyhedron = false;
139  l.unlock();
140  }
141  return fpPolyhedron;
142 }
