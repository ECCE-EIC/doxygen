39 DetectorMessenger::DetectorMessenger(DetectorConstruction * Det)
40 :G4UImessenger(),fDetector(Det),
41  fTestemDir(0),
42  fDetDir(0),
43  fMaterCmd(0),
44  fRadiusCmd(0)
45 {
46  fTestemDir = new G4UIdirectory("/mfp/");
47  fTestemDir->SetGuidance("Detector control.");
48 
49  fDetDir = new G4UIdirectory("/mfp/det/");
50  fDetDir->SetGuidance("Detector construction commands");
51 
52  fMaterCmd = new G4UIcmdWithAString("/mfp/det/setMat",this);
53  fMaterCmd->SetGuidance("Select material of the sphere.");
54  fMaterCmd->SetParameterName("choice",false);
55  fMaterCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
56  fMaterCmd->SetToBeBroadcasted(false);
57 
58  fRadiusCmd = new G4UIcmdWithADoubleAndUnit("/mfp/det/setRadius",this);
59  fRadiusCmd->SetGuidance("Set radius of the absorber");
60  fRadiusCmd->SetParameterName("Radius",false);
61  fRadiusCmd->SetRange("Radius>0.");
62  fRadiusCmd->SetUnitCategory("Length");
63  fRadiusCmd->AvailableForStates(G4State_PreInit,G4State_Idle);
64  fRadiusCmd->SetToBeBroadcasted(false);
65 }
