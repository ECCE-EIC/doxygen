43 void
44 G4TrajectoryDrawByOriginVolume::Draw(const G4VTrajectory& traj, const G4bool& /*visible*/) const
45 {
46  G4VTrajectoryPoint* aTrajectoryPoint = traj.GetPoint(0);
47  assert (0 != aTrajectoryPoint);
48 
49  G4Colour colour(fDefault);
50 
51  G4Navigator* navigator =
52  G4TransportationManager::GetTransportationManager()->GetNavigatorForTracking();
53 
54  G4VPhysicalVolume* volume = navigator->LocateGlobalPointAndSetup
55  (aTrajectoryPoint->GetPosition(), nullptr,false,true);
56 
57  // Logical volumes form basis.
58  G4LogicalVolume* logicalVolume = volume->GetLogicalVolume();
59  assert (0 != logicalVolume);
60 
61  G4String logicalName = logicalVolume->GetName();
62  fMap.GetColour(logicalName, colour);
63 
64  // Override with physical volume colouring if it exists
65  G4String physicalName = volume->GetName();
66  fMap.GetColour(physicalName, colour);
67 
68  G4VisTrajContext myContext(GetContext());
69 
70  myContext.SetLineColour(colour);
71 
72  if (GetVerbose()) {
73  G4cout<<"G4TrajectoryDrawByOriginVolume drawer named "<<Name();
74  G4cout<<", drawing trajectory originating in logical volume, "<<logicalName;
75  G4cout<<", physical volume "<<physicalName<<", with configuration:"<<G4endl;
76  myContext.Print(G4cout);
77  }
78 
79  G4TrajectoryDrawerUtils::DrawLineAndPoints(traj, myContext);
80 }
