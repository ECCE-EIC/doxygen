105 G4double G4UPiNuclearCrossSection::Interpolate(
106  G4int Z, G4int A, G4double e, const G4PhysicsTable* table) const
107 {
108  G4double res = 0.0;
109  G4double ekin = std::max(e, elow);
110  G4int iz = std::min(Z, 92);
111  G4int idx = idxZ[iz];
112  size_t jdx = (size_t)(std::max(ekin - elow, 0.0)*0.06);
113  //G4cout << "Interpolate: Z= " << iz << " A= " << A << " idx= " << idx
114  // << " jdx= " << jdx << " Ekin= " << ekin << G4endl;
115  if(idx < 0 || 2 == iz) {
116  res = ((*table)[std::abs(idx)])->Value(ekin, jdx);
117  //G4cout << "1: jdx= " << jdx << G4endl;
118  } else {
119  G4int iz2 = theZ[idx];
120  G4double x2 = (((*table)[idx])->Value(ekin, jdx))*APower[iz]/APower[iz2];
121  //G4cout << "2: jdx= " << jdx << G4endl;
122  G4int iz1 = theZ[idx-1];
123  G4double x1 = (((*table)[idx-1])->Value(ekin, jdx))*APower[iz]/APower[iz1];
124  G4double w1 = ((G4double)A - theA[idx-1])/(theA[idx] - theA[idx-1]);
125  res = w1*x2 + (1.0 - w1)*x1;
126  }
127  //G4cout << " res(nb)= " << res/CLHEP::barn << G4endl;
128  return res;
129 }

131 void G4UPiNuclearCrossSection::AddDataSet(const G4String& p,
132  const G4double* tot,
133  const G4double* in,
134  const G4double* e,
135  G4int n)
136 {
137  G4LPhysicsFreeVector* pvin = new G4LPhysicsFreeVector(n,e[0]*GeV,e[n-1]*GeV);
138  pvin->SetSpline(true);
139  G4LPhysicsFreeVector* pvel = new G4LPhysicsFreeVector(n,e[0]*GeV,e[n-1]*GeV);
140  pvel->SetSpline(true);
141  for(G4int i=0; i<n; ++i) {
142  pvin->PutValues(i,e[i]*GeV,in[i]*millibarn);
143  pvel->PutValues(i,e[i]*GeV,std::max(0.0,(tot[i]-in[i])*millibarn));
144  }
145  if(p == "pi+") {
146  piPlusInelastic->push_back(pvin);
147  piPlusElastic->push_back(pvel);
148  } else {
149  piMinusInelastic->push_back(pvin);
150  piMinusElastic->push_back(pvel);
151  }
152 }

169 void G4UPiNuclearCrossSection::BuildPhysicsTable(const G4ParticleDefinition& p)
170 {
171  //G4cout << "G4UPiNuclearCrossSection::BuildPhysicsTable for "
172  // << p.GetParticleName() << " " << piPlusElastic << G4endl;
173  if(piPlusElastic) { return; }
174 
175  if(&p != piPlus && &p != piMinus) {
176  G4ExceptionDescription ed;
177  ed << "This cross section is applicable only to pions and not to "
178  << p.GetParticleName() << G4endl;
179  G4Exception("G4UPiNuclearCrossSection::BuildPhysicsTable", "had001",
180  FatalException, ed);
181  return;
182  }
183 
184  if(!piPlusElastic) {
185 #ifdef G4MULTITHREADED
186  G4MUTEXLOCK(&pionUXSMutex);
187  if(!piPlusElastic) {
188 #endif
189  isMaster = true;
190 #ifdef G4MULTITHREADED
191  }
192  G4MUTEXUNLOCK(&pionUXSMutex);
193 #endif
194  }
195  if(!isMaster) { return; }
196 
197  G4NistManager* nist = G4NistManager::Instance();
198  G4Pow* g4pow = G4Pow::GetInstance();
199  for(G4int i=0; i<NZ; ++i) {
200  theA[i] = nist->GetAtomicMassAmu(theZ[i]);
201  }
202  for(G4int i=1; i<93; ++i) {
203  APower[i] = g4pow->powA(nist->GetAtomicMassAmu(i),aPower);
204  }
205  G4int idx = 1;
206  for(G4int i=3; i<93; ++i) {
207  if(theZ[idx] == i) {
208  idxZ[i] = -idx;
209  ++idx;
210  } else {
211  idxZ[i] = idx;
212  }
213  }
214 
215  piPlusElastic = new G4PhysicsTable();
216  piPlusInelastic = new G4PhysicsTable();
217  piMinusElastic = new G4PhysicsTable();
218  piMinusInelastic = new G4PhysicsTable();
219 
220  LoadData();
221  //G4cout << "### Initialisation done " << G4endl;
222 }
