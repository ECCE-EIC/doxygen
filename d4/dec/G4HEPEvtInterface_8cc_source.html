63 void G4HEPEvtInterface::GeneratePrimaryVertex(G4Event* evt)
64 {
65  G4int NHEP = 0; // number of entries
66  if (inputFile.is_open()) {
67  inputFile >> NHEP;
68  }
69  else {
70  G4Exception("G4HEPEvtInterface::G4HEPEvtInterface","Event0201",FatalException,
71  "G4HEPEvtInterface:: cannot open file.");
72  }
73  if( inputFile.eof() )
74  {
75  G4Exception("G4HEPEvtInterface::GeneratePrimaryVertex","Event0202",
76  RunMustBeAborted,"End-Of-File : HEPEvt input file -- no more event to read!");
77  return;
78  }
79 
80  if(vLevel > 0)
81  {
82  G4cout << "G4HEPEvtInterface - reading " << NHEP << " HEPEvt particles from "
83  << fileName << "." << G4endl;
84  }
85  for( G4int IHEP=0; IHEP<NHEP; IHEP++ )
86  {
87  G4int ISTHEP; // status code
88  G4int IDHEP; // PDG code
89  G4int JDAHEP1; // first daughter
90  G4int JDAHEP2; // last daughter
91  G4double PHEP1; // px in GeV
92  G4double PHEP2; // py in GeV
93  G4double PHEP3; // pz in GeV
94  G4double PHEP5; // mass in GeV
95 
96  inputFile >> ISTHEP >> IDHEP >> JDAHEP1 >> JDAHEP2
97  >> PHEP1 >> PHEP2 >> PHEP3 >> PHEP5;
98  if( inputFile.eof() )
99  {
100  G4Exception("G4HEPEvtInterface::GeneratePrimaryVertex","Event0203",
101  FatalException,"Unexpected End-Of-File in the middle of an event");
102  }
103  if(vLevel > 1)
104  {
105  G4cout << " " << ISTHEP << " " << IDHEP << " " << JDAHEP1 << " " << JDAHEP2
106  << " " << PHEP1 << " " << PHEP2 << " " << PHEP3 << " " << PHEP5
107  << G4endl;
108  }
109 
110  // create G4PrimaryParticle object
111  G4PrimaryParticle* particle
112  = new G4PrimaryParticle( IDHEP );
113  particle->SetMass( PHEP5*GeV );
114  particle->SetMomentum(PHEP1*GeV, PHEP2*GeV, PHEP3*GeV );
115 
116  // create G4HEPEvtParticle object
117  G4HEPEvtParticle* hepParticle
118  = new G4HEPEvtParticle( particle, ISTHEP, JDAHEP1, JDAHEP2 );
119 
120  // Store
121  HPlist.push_back( hepParticle );
122  }
123 
124  // check if there is at least one particle
125  if( HPlist.size() == 0 ) return;
126 
127  // make connection between daughter particles decayed from
128  // the same mother
129  for( size_t i=0; i<HPlist.size(); i++ )
130  {
131  if( HPlist[i]->GetJDAHEP1() > 0 ) // it has daughters
132  {
133  G4int jda1 = HPlist[i]->GetJDAHEP1()-1; // FORTRAN index starts from 1
134  G4int jda2 = HPlist[i]->GetJDAHEP2()-1; // but C++ starts from 0.
135  G4PrimaryParticle* mother = HPlist[i]->GetTheParticle();
136  for( G4int j=jda1; j<=jda2; j++ )
137  {
138  G4PrimaryParticle* daughter = HPlist[j]->GetTheParticle();
139  if(HPlist[j]->GetISTHEP()>0)
140  {
141  mother->SetDaughter( daughter );
142  HPlist[j]->Done();
143  }
144  }
145  }
146  }
147 
148  // create G4PrimaryVertex object
149  G4PrimaryVertex* vertex = new G4PrimaryVertex(particle_position,particle_time);
150 
151  // put initial particles to the vertex
152  for( size_t ii=0; ii<HPlist.size(); ii++ )
153  {
154  if( HPlist[ii]->GetISTHEP() > 0 ) // ISTHEP of daughters had been
155  // set to negative
156  {
157  G4PrimaryParticle* initialParticle = HPlist[ii]->GetTheParticle();
158  vertex->SetPrimary( initialParticle );
159  }
160  }
161 
162  // clear G4HEPEvtParticles
163  //HPlist.clearAndDestroy();
164  for(size_t iii=0;iii<HPlist.size();iii++)
165  { delete HPlist[iii]; }
166  HPlist.clear();
167 
168  // Put the vertex to G4Event object
169  evt->AddPrimaryVertex( vertex );
170 }
