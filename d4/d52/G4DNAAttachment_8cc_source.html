58 void G4DNAAttachment::InitialiseProcess(const G4ParticleDefinition* p)
59 {
60  if(!isInitialised)
61  {
62  isInitialised = true;
63  SetBuildTableFlag(false);
64 
65  G4String name = p->GetParticleName();
66 
67  if(name == "e-" )
68  {
69  if(!EmModel())
70  {
71  SetEmModel(new G4DNAMeltonAttachmentModel);
72  EmModel()->SetLowEnergyLimit(4.*eV);
73  EmModel()->SetHighEnergyLimit(13.*eV);
74  }
75  AddEmModel(1, EmModel());
76  }
77  else if(name == "e+" )
78  {
79  if(!EmModel())
80  {
81  SetEmModel(new G4LEPTSAttachmentModel);
82  EmModel()->SetLowEnergyLimit(1.*eV);
83  EmModel()->SetHighEnergyLimit(1.*MeV);
84  }
85  AddEmModel(1, EmModel());
86  }
87  }
88 }
