153 inline
154 G4PhysicsVector* G4ElementData::GetComponentDataByID(G4int Z, G4int id)
155 {
156  G4PhysicsVector* v = 0;
157  for(size_t i=0; i<compLength[Z]; ++i) {
158  if(id == (compID[Z])[i]) {
159  v = (compData[Z])[i];
160  break;
161  }
162  }
163  return v;
164 }
