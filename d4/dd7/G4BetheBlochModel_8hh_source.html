170 inline void G4BetheBlochModel::SetParticle(const G4ParticleDefinition* p)
171 {
172  if(particle != p) {
173  particle = p;
174  if(p->GetBaryonNumber() > 3 || p->GetPDGCharge() > CLHEP::eplus)
175  { isIon = true; }
176  SetupParameters();
177  }
178 }

182 inline void G4BetheBlochModel::SetGenericIon(const G4ParticleDefinition* p)
183 {
184  if(p && p->GetParticleName() == "GenericIon") { isIon = true; }
185 }
