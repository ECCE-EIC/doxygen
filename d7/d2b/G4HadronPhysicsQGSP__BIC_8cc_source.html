91 G4HadronPhysicsQGSP_BIC::G4HadronPhysicsQGSP_BIC(const G4String& name, G4bool)
92  : G4VPhysicsConstructor(name)
93 {
94  QuasiElasticFTF= false; // Use built-in quasi-elastic (not add-on)
95  QuasiElasticQGS= true; // For QGS, it must use it.
96  minQGSP_proton = minQGSP_neutron = minQGSP_pik =
97  G4HadronicParameters::Instance()->GetMinEnergyTransitionQGS_FTF();
98  maxFTFP_proton = maxFTFP_neutron = maxFTFP_pik =
99  G4HadronicParameters::Instance()->GetMaxEnergyTransitionQGS_FTF();
100  minFTFP_proton = minFTFP_neutron = minFTFP_pik =
101  G4HadronicParameters::Instance()->GetMinEnergyTransitionFTF_Cascade();
102  maxBIC_proton = maxBIC_neutron = maxBERT_pik =
103  G4HadronicParameters::Instance()->GetMaxEnergyTransitionFTF_Cascade();
104 }

115 void G4HadronPhysicsQGSP_BIC::Neutron()
116 {
117  auto neu = new G4NeutronBuilder;
118  AddBuilder(neu);
119  auto qgs = new G4QGSPNeutronBuilder(QuasiElasticQGS);
120  AddBuilder(qgs);
121  qgs->SetMinEnergy(minQGSP_neutron);
122  neu->RegisterMe(qgs);
123  auto ftf = new G4FTFPNeutronBuilder(QuasiElasticFTF);
124  AddBuilder(ftf);
125  ftf->SetMinEnergy(minFTFP_neutron);
126  ftf->SetMaxEnergy(maxFTFP_neutron);
127  neu->RegisterMe(ftf);
128  auto bic = new G4BinaryNeutronBuilder;
129  AddBuilder(bic);
130  bic->SetMaxEnergy(maxBIC_neutron);
131  neu->RegisterMe(bic);
132  neu->Build();
133 }

155 void G4HadronPhysicsQGSP_BIC::Pion()
156 {
157  auto pik = new G4PiKBuilder;
158  AddBuilder(pik);
159  auto qgs = new G4QGSPPiKBuilder(QuasiElasticQGS);
160  AddBuilder(qgs);
161  qgs->SetMinEnergy(minQGSP_pik);
162  pik->RegisterMe(qgs);
163  auto ftf = new G4FTFPPiKBuilder(QuasiElasticFTF);
164  AddBuilder(ftf);
165  ftf->SetMaxEnergy(maxFTFP_pik);
166  ftf->SetMinEnergy(minFTFP_pik);
167  pik->RegisterMe(ftf);
168  auto bert = new G4BertiniPiKBuilder;
169  AddBuilder(bert);
170  bert->SetMaxEnergy(maxBERT_pik);
171  pik->RegisterMe(bert);
172  pik->Build();

215 void G4HadronPhysicsQGSP_BIC::ExtraConfiguration()
216 {
217  // --- Neutrons ---
218  const G4ParticleDefinition* neutron = G4Neutron::Neutron();
219  G4HadronicProcess* inel = G4PhysListUtil::FindInelasticProcess(neutron);
220  if(inel) { inel->AddDataSet(new G4NeutronInelasticXS()); }
221  G4HadronicProcess* capture = G4PhysListUtil::FindCaptureProcess(neutron);
222  if (capture) {
223  capture->RegisterMe(new G4NeutronRadCapture());
224  }
225 }
