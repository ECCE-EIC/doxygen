40 G4int CCalEcal::readFile() {
42  //Let's open the file
43  G4cout << " ==> Opening file " << File() << " to read elements..."
44  << G4endl;
45 
46  std::ifstream is;
47  G4bool ok = openGeomFile(is, pathName, File());
48  if (!ok)
49  return 0;
50 
51  // Find *DO CrystalMatrix
52  findDO(is, G4String("CrystalMatrix"));
53 
54  //Let's read overall box dimensions and positions
55  readName(is,genMat);
56  is >> widBox >> lengBox >> xpos >> ypos >> zpos >> thetaX >> phiX
57  >> thetaY >> phiY >> thetaZ >> phiZ >> jump;
58 #ifdef debug
59  G4cout << tab << "General material: " << genMat << "\tBox dimensions "
60  << widBox << ", " << lengBox << G4endl;
61  G4cout << tab << "Positioned at (" << xpos << ", " << ypos << ", " << zpos
62  << ") with rotation (" << thetaX << ", " << phiX << ", " << thetaY
63  << ", " << phiY << ", " << thetaZ << ", " << phiZ << ")" << G4endl;
64 #endif
65 
66  //Then the layer positions
67  G4int i=0;
68  readName(is,layMat);
69  is >> layNum >> layRadius >> layAngle >> lengFront;
70  for (i=0; i<5; i++)
71  is >> layPar[i];
72 #ifdef debug
73  G4cout << tab << "Layer material: " << layMat << " Number " << layNum
74  << " Radius " << layRadius << " Angle " << layAngle/deg
75  << " front dist " << lengFront << " Parameters ";
76  for (i=0; i<5; i++)
77  G4cout << layPar[i] << " ";
78  G4cout << G4endl;
79 #endif
80 
81  //Then the crystal positions
82  readName(is,crystMat);
83  is >> crystNum >> crystLength >> crystTol;
84  for (i=0; i<5; i++)
85  is >> crystPar[i];
86 #ifdef debug
87  G4cout << tab << "Crystal material: " << crystMat << " Number " << crystNum
88  << " Length " << crystLength << " Tolerance " << crystTol
89  << " Parameters ";
90  for (i=0; i<5; i++)
91  G4cout << crystPar[i] << " ";
92  G4cout << G4endl;
93 #endif
94 
95  //Then the support material
96  readName(is,suppMat);
97  is >> dxSupp >> dySupp >> dzSupp >> distSupp >> jump;
98 #ifdef debug
99  G4cout << tab << "Support material: " << suppMat << " Dimensions " << dxSupp
100  << ", " << dySupp << ", " << dzSupp << " Distance " << distSupp << G4endl;
101 #endif
102 
104  // Close the file
105  G4cout << " ==> Closing file " << File() << G4endl;
106  is.close();
107 
108  return 1;
