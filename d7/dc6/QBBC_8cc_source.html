64 QBBC::QBBC( G4int ver, const G4String&)
65 {
66  G4cout << "<<< Reference Physics List QBBC " <<G4endl;
67 
68  defaultCutValue = 0.7*mm;
69  SetVerboseLevel(ver);
70 
71  // EM Physics
72  RegisterPhysics( new G4EmStandardPhysics(ver) );
73 
74  // Synchroton Radiation & GN Physics
75  RegisterPhysics( new G4EmExtraPhysics(ver) );
76 
77  // Decays
78  RegisterPhysics( new G4DecayPhysics(ver) );
79 
80  // Hadron Physics
81  RegisterPhysics( new G4HadronElasticPhysicsXS(ver) );
82 
83  RegisterPhysics( new G4StoppingPhysics(ver) );
84 
85  RegisterPhysics( new G4IonPhysicsXS(ver) );
86 
87  RegisterPhysics( new G4IonElasticPhysics(ver) );
88 
89  RegisterPhysics( new G4HadronInelasticQBBC(ver));
90 
91  // Neutron tracking cut
92  RegisterPhysics( new G4NeutronTrackingCut(ver) );
93 }
