54 void SteppingAction::UserSteppingAction(const G4Step* step)
55 {
56  G4double edep = step->GetTotalEnergyDeposit();
57  if (edep <= 0.) return;
58 
59  //total energy deposit in absorber
60  //
61  fEventAction->AddEdep(edep);
62 
63  //longitudinal profile of deposited energy
64  //
65  G4ThreeVector prePoint = step->GetPreStepPoint() ->GetPosition();
66  G4ThreeVector postPoint = step->GetPostStepPoint()->GetPosition();
67  G4ThreeVector point = prePoint + G4UniformRand()*(postPoint - prePoint);
68  if (step->GetTrack()->GetDefinition()->GetPDGCharge() == 0) point = postPoint;
69  G4double r = point.mag();
70  G4AnalysisManager* analysisManager = G4AnalysisManager::Instance();
71  analysisManager->FillH1(1, r, edep);
72 
73  Run* run
74  = static_cast<Run*>(G4RunManager::GetRunManager()->GetNonConstCurrentRun());
75 
76  G4double r0 = run->GetCsdaRange();
77  if (r0 > 0.) analysisManager->FillH1(8, r/r0, edep);
78 
79  //step size of primary particle or charged secondaries
80  //
81  G4double steplen = step->GetStepLength();
82  const G4Track* track = step->GetTrack();
83  if (track->GetTrackID() == 1) analysisManager->FillH1(4, steplen);
84  else if (track->GetDefinition()->GetPDGCharge() != 0.)
85  analysisManager->FillH1(7, steplen);
86 }
