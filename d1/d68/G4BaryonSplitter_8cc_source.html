64 G4bool G4BaryonSplitter::
65 SplitBarion(G4int PDGCode, G4int* q_or_qqbar, G4int* qbar_or_qq)
66 {
67  const G4SPBaryon * aBaryon = theBaryons.GetBaryon(G4ParticleTable::GetParticleTable()->FindParticle(PDGCode));
68 
69  if(aBaryon==NULL)
70  {
71  return FALSE;
72  } else {
73  aBaryon->SampleQuarkAndDiquark(*q_or_qqbar, *qbar_or_qq);
74  return TRUE;
75  }
76 }

87 // Find rest diquark in given barion after quark - antiquark annihilation
88 G4bool G4BaryonSplitter::
89 FindDiquark(G4int PDGCode, G4int Quark, G4int* Diquark)
90 {
91  const G4SPBaryon * aBaryon = theBaryons.GetBaryon(G4ParticleTable::GetParticleTable()->FindParticle(PDGCode));
92  if(aBaryon)
93  {
94  aBaryon->FindDiquark(Quark, *Diquark);
95  return true;

97  return false;
