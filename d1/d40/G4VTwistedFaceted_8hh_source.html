297 inline
298 G4double G4VTwistedFaceted::GetCubicVolume()
299 {
300  if(fCubicVolume != 0.) ;
301  else fCubicVolume = 2 * fDz
302  * ( ( fDx1 + fDx2 ) * fDy1 + ( fDx3 + fDx4 ) * fDy2 );
303  return fCubicVolume;
304 }

314 inline
315 G4double G4VTwistedFaceted::GetValueA(G4double phi) const
316 {
317  return ( fDx4 + fDx2 + ( fDx4 - fDx2 ) * ( 2 * phi ) / fPhiTwist ) ;
318 }
319 
320 inline
321 G4double G4VTwistedFaceted::GetValueD(G4double phi) const
322 {
323  return ( fDx3 + fDx1 + ( fDx3 - fDx1 ) * ( 2 * phi ) / fPhiTwist ) ;
324 }
325 
326 inline
327 G4double G4VTwistedFaceted::GetValueB(G4double phi) const
328 {
329  return ( fDy2 + fDy1 + ( fDy2 - fDy1 ) * ( 2 * phi ) / fPhiTwist ) ;
330 }

332 inline
333 G4double G4VTwistedFaceted::Xcoef(G4double u, G4double phi, G4double ftg) const
334 {
335  return GetValueA(phi)/2. + (GetValueD(phi)-GetValueA(phi))/4.
336  - u*( ( GetValueD(phi)-GetValueA(phi) ) / ( 2 * GetValueB(phi) ) - ftg );
337 }
