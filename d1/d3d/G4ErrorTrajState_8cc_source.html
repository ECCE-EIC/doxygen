83 //--------------------------------------------------------------------------
84 void G4ErrorTrajState::BuildCharge()
85 {
86  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
87  G4ParticleDefinition* particle = particleTable->FindParticle(fParticleType);
88  if( particle == 0)
89  {
90  std::ostringstream message;
91  message << "Particle type not defined: " << fParticleType;
92  G4Exception( "G4ErrorTrajState::BuildCharge()", "GEANT4e-error",
93  FatalException, message);
94  }
95  else
96  {
97  fCharge = particle->GetPDGCharge();
98  }
99 }
