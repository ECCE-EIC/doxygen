105 inline G4double
106 G4UPiNuclearCrossSection::GetElasticCrossSection(
107  const G4DynamicParticle* dp, G4int Z, G4int A) const
108 {
109  const G4PhysicsTable* table =
110  (dp->GetDefinition() == piPlus) ? piPlusElastic : piMinusElastic;
