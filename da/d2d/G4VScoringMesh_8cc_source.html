225 G4VPrimitiveScorer * G4VScoringMesh::GetPrimitiveScorer(const G4String & name) {
226  if(!fMFD) return nullptr;
227 
228  G4int nps = fMFD->GetNumberOfPrimitives();
229  for(G4int i = 0; i < nps; i++) {
230  G4VPrimitiveScorer * prs = fMFD->GetPrimitive(i);
231  if(name == prs->GetName()) return prs;
232  }
233 
234  return nullptr;
235 }
