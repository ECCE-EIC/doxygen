176 inline
177  G4String G4ExcitedMesonConstructor::GetName(G4int iIso3,
178  G4int iState,
179  G4int iType)
180 {
181  G4String particle = name[iState][iType];
182  if (iType == TPi) {
183  if ( iIso3 == +2 ){
184  particle += "+";
185  } else if ( iIso3 == -2 ){
186  particle += "-";
187  } else {
188  particle += "0";
189  }
190  } else if (iType == TK) {
191  if ( iIso3 == +1 ){
192  particle += "+";
193  } else if ( iIso3 == -1 ){
194  particle += "0";
195  }
196  } else if (iType == TAntiK) {
197  if ( iIso3 == +1 ){
198  particle += "0";
199  particle = "anti_" + particle;
200  } else if ( iIso3 == -1 ){
201  particle += "-";
202  }
203  }
204  return particle;
205 }
