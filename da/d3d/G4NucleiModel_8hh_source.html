120  G4double getPotential(G4int ip, G4int izone) const {
121  if (ip == 9 || ip < 0) return 0.0; // Photons and leptons
122  G4int ip0 = ip < 3 ? ip - 1 : 2;
123  if (ip > 10 && ip < 18) ip0 = 3;
124  if (ip > 20) ip0 = 4;
