318  BasicVector3D<T> orthogonal() const {
319  T dx = x() < 0 ? -x() : x();
320  T dy = y() < 0 ? -y() : y();
321  T dz = z() < 0 ? -z() : z();
322  if (dx < dy) {
323  return dx < dz ?
324  BasicVector3D<T>(0,z(),-y()) : BasicVector3D<T>(y(),-x(),0);
325  }else{
326  return dy < dz ?
327  BasicVector3D<T>(-z(),0,x()) : BasicVector3D<T>(y(),-x(),0);
328  }
329  }
