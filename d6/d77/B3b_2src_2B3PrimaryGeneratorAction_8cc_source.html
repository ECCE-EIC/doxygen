71 void B3PrimaryGeneratorAction::GeneratePrimaries(G4Event* anEvent)
72 {
73  G4ParticleDefinition* particle = fParticleGun->GetParticleDefinition();
74  if (particle == G4ChargedGeantino::ChargedGeantino()) {
75  //fluorine
76  G4int Z = 9, A = 18;
77  G4double ionCharge = 0.*eplus;
78  G4double excitEnergy = 0.*keV;
79 
80  G4ParticleDefinition* ion
81  = G4IonTable::GetIonTable()->GetIon(Z,A,excitEnergy);
82  fParticleGun->SetParticleDefinition(ion);
83  fParticleGun->SetParticleCharge(ionCharge);
84  }
85 
86  // randomized position
87  //
90  G4double x0 = 4*cm, y0 = 4*cm, z0 = 4*cm;
91  G4double dx0 = 1*cm, dy0 = 1*cm, dz0 = 1*cm;
92  x0 += dx0*(G4UniformRand()-0.5);
93  y0 += dy0*(G4UniformRand()-0.5);
94  z0 += dz0*(G4UniformRand()-0.5);
95  fParticleGun->SetParticlePosition(G4ThreeVector(x0,y0,z0));
96 
97  //create vertex
98  //
99  fParticleGun->GeneratePrimaryVertex(anEvent);
100 }
