142 inline void G4PrimaryVertex::SetPrimary(G4PrimaryParticle * pp)
143 {
144  if(theParticle == 0) { theParticle = pp; }
145  else { theTail->SetNext(pp); }
146  theTail = pp;
147  numberOfParticle++;
148 }
