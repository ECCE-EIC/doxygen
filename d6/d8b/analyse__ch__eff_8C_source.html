11  Double_t vEfficiency = h1->Integral(vBinMin+1,vBinMax-1);

13  vEfficiency += h1->GetBinContent(vBinMax)*(-h1->GetBinLowEdge(vBinMax)+vXMax)/h1->GetBinWidth(vBinMax);
14  vEfficiency /= h1->Integral(0,-1);
